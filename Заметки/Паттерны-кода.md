# Паттерны кода в Oshiqona

## 1. Smart/Dumb компоненты

### Smart (Container) компонент
```typescript
// Знает о сервисах, управляет состоянием
@Component({
  selector: 'app-chats',
  template: `
    <app-chat-list
      [chats]="chats$ | async"
      (chatSelected)="onChatSelected($event)"
    />
  `,
  changeDetection: ChangeDetectionStrategy.OnPush,
  standalone: true,
})
export class ChatsComponent {
  chats$ = this.chatService.getChats();

  constructor(private chatService: ChatService) {}

  onChatSelected(chatId: string) {
    this.router.navigate(['/chats', chatId]);
  }
}
```

### Dumb (Presentational) компонент
```typescript
// Только отображение, Input/Output
@Component({
  selector: 'app-chat-list',
  template: `
    <div *ngFor="let chat of chats" (click)="chatSelected.emit(chat.id)">
      {{ chat.userName }}
    </div>
  `,
  changeDetection: ChangeDetectionStrategy.OnPush,
  standalone: true,
})
export class ChatListComponent {
  @Input() chats: Chat[] = [];
  @Output() chatSelected = new EventEmitter<string>();
}
```

## 2. Reactive Forms паттерн

```typescript
@Component({...})
export class ProfileEditComponent {
  form = this.fb.group({
    name: ['', [Validators.required, Validators.minLength(2)]],
    email: ['', [Validators.required, Validators.email]],
    description: ['', Validators.maxLength(500)],
  });

  constructor(
    private fb: FormBuilder,
    private userService: CurrentUserService,
  ) {}

  ngOnInit() {
    // Заполнить форму данными
    this.userService.getCurrentUser().subscribe(user => {
      this.form.patchValue(user);
    });
  }

  onSubmit() {
    if (this.form.invalid) return;
    this.userService.updateProfile(this.form.value).subscribe();
  }

  // Геттеры для удобного доступа в шаблоне
  get nameControl() { return this.form.get('name')!; }
  get emailControl() { return this.form.get('email')!; }
}
```

## 3. Async Pipe паттерн

```typescript
// Компонент
@Component({
  template: `
    <!-- Правильно: async pipe управляет подпиской -->
    <div *ngFor="let chat of chats$ | async">
      {{ chat.userName }}
    </div>

    <!-- Правильно: обработка состояний -->
    <ng-container *ngIf="chats$ | async as chats; else loading">
      <app-chat-list [chats]="chats" />
    </ng-container>
    <ng-template #loading>
      <tui-loader />
    </ng-template>
  `,
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class ChatsComponent {
  chats$ = this.chatService.getChats();
  constructor(private chatService: ChatService) {}
}
```

## 4. Destroy паттерн (Angular 16+)

```typescript
@Component({...})
export class ChatDetailComponent {
  private destroy$ = new Subject<void>();

  ngOnInit() {
    this.chatHub.onMessageReceived()
      .pipe(takeUntil(this.destroy$))
      .subscribe(message => {
        // обработка
      });
  }

  ngOnDestroy() {
    this.destroy$.next();
    this.destroy$.complete();
  }
}

// Или через takeUntilDestroyed (Angular 16+)
@Component({...})
export class ChatDetailComponent {
  private destroyRef = inject(DestroyRef);

  ngOnInit() {
    this.chatHub.onMessageReceived()
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe(message => {
        // обработка
      });
  }
}
```

## 5. Error Handling паттерн

```typescript
// В сервисе
getData(): Observable<Data[]> {
  return this.http.get<Data[]>('/api/data').pipe(
    catchError(error => {
      // Логировать ошибку
      console.error('Ошибка загрузки данных:', error);
      // Вернуть пустой массив или пробросить ошибку
      return throwError(() => new Error('Не удалось загрузить данные'));
    })
  );
}

// В компоненте
@Component({
  template: `
    <ng-container *ngIf="data$ | async as data; else errorTpl">
      <app-data-list [data]="data" />
    </ng-container>
    <ng-template #errorTpl>
      <p>Ошибка загрузки</p>
    </ng-template>
  `,
})
export class DataComponent {
  data$ = this.service.getData().pipe(
    catchError(() => of([]))  // Fallback к пустому массиву
  );
}
```

## 7. Signal паттерн (Angular 17+)

```typescript
@Component({...})
export class CounterComponent {
  // Signal
  count = signal(0);

  // Computed signal
  doubled = computed(() => this.count() * 2);

  // Effect
  constructor() {
    effect(() => {
      console.log('Count changed:', this.count());
    });
  }

  increment() {
    this.count.update(v => v + 1);
  }
}
```
